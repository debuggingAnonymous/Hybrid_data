eval1 

{

  VALUE *l;

  VALUE *r;



#ifdef EVAL_TRACE

  trace ("eval1");

#endif

  l = eval2 (evaluate);

  while (1)

    {

      if (nextarg ("&"))

	{

	  r = eval2 (evaluate & ~ null (l));

	  if (null (l) || null (r))

	    {

	      freev (l);

	      freev (r);

	      l = int_value (0);

	    }

	  else

	    freev (r);

	}

      else

	return l;

    }

}
