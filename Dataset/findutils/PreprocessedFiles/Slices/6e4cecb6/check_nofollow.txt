check_nofollow 

{

  struct utsname uts;

  float release;



  if (0 == uname(&uts))

    {



      release = atof(uts.release);



      if (0 == strcmp("Linux", uts.sysname))

 {



   return release >= 2.2;

 }

      else if (0 == strcmp("FreeBSD", uts.sysname))

 {



   return release >= 3.1;

 }

    }





  return 1;

}
